/// Parse source and return true if there are errors.
fn parse_has_errors(src: &str) -> bool {
    let tokens: Vec<_> = Token::lexer(src)
        .spanned()
        .map(|(tok, span)| match tok {
            Ok(t) => (t, Span::new(span.start, span.end)),
            Err(()) => (Token::Error, Span::new(span.start, span.end)),
        })
        .collect();

    let len = src.len();
    let stream = chumsky::input::Stream::from_iter(tokens).map(Span::new(len, len), |(t, s)| (t, s));
    let (_, errors) = crate::parser::parse(src).into_output_errors();
    !errors.is_empty()
}

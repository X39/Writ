```rust
/// Push namespace segments onto the stack.
pub fn push_namespace(&mut self, segments: &[String]) {
    self.namespace_stack.extend(segments.iter().cloned());
}

/// Pop `count` segments from the namespace stack.
pub fn pop_namespace(&mut self, count: usize) {
    let new_len = self.namespace_stack.len().saturating_sub(count);
    self.namespace_stack.truncate(new_len);
}

/// Set the namespace stack to the given segments (for declarative namespaces).
pub fn set_namespace(&mut self, segments: Vec<String>) {
    self.namespace_stack = segments;
}

/// Get the current namespace as a "::" joined string. Returns empty string if no namespace.
pub fn current_namespace(&self) -> String {
    self.namespace_stack.join("::")
}
```

**1. DLG-02: Preserve interpolation slot names in raw_text_content()**

The `raw_text_content()` function (line 423-432) replaces ALL `DlgTextSegment::Expr` with `"{expr}"`. This loses the original expression text like `{name}` or `{player.name}`.

The `DlgTextSegment::Expr` variant contains `Box<Spanned<Expr>>`. The inner expression's span covers the expression source text. However, at lowering time we don't have the original source text string, so we can't slice into it using the span.

**Approach**: Reconstruct a textual representation from the CST expression. For simple cases (`{name}` -> `Expr::Ident("name")`), return the identifier. For member access (`{player.name}` -> `Expr::MemberAccess`), reconstruct `player.name`. For complex expressions, provide a best-effort textual form.

Add a helper function that converts a CST `Expr` to a representative string for localization content:

```rust
/// Reconstruct a textual representation of a CST expression for localization slot content.
/// Used to preserve interpolation slot identities in raw text (e.g., `{name}` stays `{name}`).
fn expr_to_slot_text(expr: &Spanned<writ_parser::cst::Expr<'_>>) -> String {
    use writ_parser::cst::Expr;
    match &expr.0 {
        Expr::Ident(name) => name.to_string(),
        Expr::MemberAccess { object, field, .. } => {
            format!("{}.{}", expr_to_slot_text(object), field)
        }
        Expr::Call { callee, .. } => {
            format!("{}(..)", expr_to_slot_text(callee))
        }
        // For other expression types, use a generic but unique representation
        // based on the expression span range to maintain uniqueness for hashing
        _ => format!("expr@{}..{}", expr.1.start, expr.1.end),
    }
}
```

Update `raw_text_content()` to use this helper:


**4. DLG-05: Speaker scope isolation in if/match branches**

**Update `lower_dlg_if()` (line 610-629):**

```rust
fn lower_dlg_if(
    dlg_if: DlgIf<'_>,
    if_span: SimpleSpan,
    state: &mut DlgLowerState,
    ctx: &mut LoweringContext,
) -> AstStmt {
    let condition = lower_expr(*dlg_if.condition, ctx);

    // Save speaker scope before then-block
    let depth = ctx.speaker_stack_depth();
    let then_block = lower_dlg_lines(&dlg_if.then_block, state, ctx);
    // Restore speaker scope after then-block
    while ctx.speaker_stack_depth() > depth {
        ctx.pop_speaker();
    }

    let else_block = lower_dlg_else(dlg_if.else_block, state, ctx);

    AstStmt::Expr {
        expr: AstExpr::If {
            condition: Box::new(condition),
            then_block,
            else_block,
            span: if_span,
        },
        span: if_span,
    }
}
```

**Update `lower_dlg_else()` (line 631-658):**

Each branch (ElseIf and Else) needs its own scope isolation:

```rust
fn lower_dlg_else<'src>(
    else_block: Option<Box<Spanned<DlgElse<'src>>>>,
    state: &mut DlgLowerState,
    ctx: &mut LoweringContext,
) -> Option<Box<AstExpr>> {
    match else_block {
        None => None,
        Some(boxed) => {
            let (dlg_else, else_span) = *boxed;
            match dlg_else {
                DlgElse::ElseIf(elif) => {
                    // ElseIf recurses into lower_dlg_if which handles its own scope isolation
                    let elif_stmt = lower_dlg_if(elif, else_span, state, ctx);
                    Some(Box::new(AstExpr::Block {
                        stmts: vec![elif_stmt],
                        span: else_span,
                    }))
                }
                DlgElse::Else(lines) => {
                    // Save/restore speaker scope for else block
                    let depth = ctx.speaker_stack_depth();
                    let stmts = lower_dlg_lines(&lines, state, ctx);
                    while ctx.speaker_stack_depth() > depth {
                        ctx.pop_speaker();
                    }
                    Some(Box::new(AstExpr::Block {
                        stmts,
                        span: else_span,
                    }))
                }
            }
        }
    }
}
```

**Update `lower_dlg_match()` (line 664-689):**

```rust
fn lower_dlg_match(
    dlg_match: DlgMatch<'_>,
    match_span: SimpleSpan,
    state: &mut DlgLowerState,
    ctx: &mut LoweringContext,
) -> AstStmt {
    let scrutinee = lower_expr(*dlg_match.scrutinee, ctx);
    let arms = dlg_match
        .arms
        .into_iter()
        .map(|(arm, arm_span)| {
            // Save speaker scope before each arm
            let depth = ctx.speaker_stack_depth();
            let body = lower_dlg_lines(&arm.body, state, ctx);
            // Restore speaker scope after each arm
            while ctx.speaker_stack_depth() > depth {
                ctx.pop_speaker();
            }
            AstMatchArm {
                pattern: lower_pattern(arm.pattern, ctx),
                body,
                span: arm_span,
            }
        })
        .collect();

    AstStmt::Expr {
        expr: AstExpr::Match {
            scrutinee: Box::new(scrutinee),
            arms,
            span: match_span,
        },
        span: match_span,
    }
}
```

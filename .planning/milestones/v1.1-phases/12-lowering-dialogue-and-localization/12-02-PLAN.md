**DLG-05: Speaker scope isolation in if/match branches**

```rust
/// Speaker set inside $ if branch doesn't leak to subsequent lines.
/// Without DLG-05 fix, @Narrator set inside the if branch leaks to the text after the if.
#[test]
fn dlg_speaker_scope_isolation_if() {
    let ast = lower_src(
        r#"dlg scene(player: Entity) {
            @player Hello.
            $ if true {
                @Narrator A narrator appears.
            }
            This should use player, not Narrator.
        }"#
    );
    insta::assert_debug_snapshot!(ast);
}
```

```rust
/// Speaker set in one $ if branch doesn't leak to else branch.
#[test]
fn dlg_speaker_scope_isolation_if_else() {
    let ast = lower_src(
        r#"dlg scene(player: Entity) {
            @player Hello.
            $ if true {
                @Narrator Branch A.
            } else {
                This should use player, not Narrator.
            }
        }"#
    );
    insta::assert_debug_snapshot!(ast);
}
```

```rust
/// Speaker set inside $ match arm doesn't leak to subsequent lines.
#[test]
fn dlg_speaker_scope_isolation_match() {
    let ast = lower_src(

            @player Hello.

                _ => {
                    @Narrator The player is sad.
                }
            }
            This should use player, not Narrator.
        }"#
    );

}
```

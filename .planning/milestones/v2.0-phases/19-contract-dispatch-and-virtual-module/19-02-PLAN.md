<objective>
Build the multi-module domain infrastructure and cross-module name resolution system. A Domain holds the virtual module plus user modules. At load time, all TypeRef/MethodRef/FieldRef entries are resolved by name matching across modules. Unresolvable references produce clear error messages.

Purpose: Cross-module resolution is required for user modules to reference writ-runtime types and contracts. The dispatch table (Plan 03) needs globally-resolved type and contract identifiers.

Output: `writ-runtime/src/domain.rs` with Domain struct and resolution logic, extended LoadedModule with resolved refs.

    ```rust
    use std::collections::HashMap;
    use crate::error::RuntimeError;
    use crate::loader::LoadedModule;

    /// Resolved cross-module type reference.
    #[derive(Debug, Clone, Copy)]
    pub struct ResolvedType {
        pub module_idx: usize,
        pub typedef_idx: usize,  // 0-based index into module's type_defs
    }

    /// Resolved cross-module method reference.
    #[derive(Debug, Clone, Copy)]
    pub struct ResolvedMethod {
        pub module_idx: usize,
        pub method_idx: usize,  // 0-based index into module's decoded_bodies
    }

    /// Resolved cross-module field reference.
    #[derive(Debug, Clone, Copy)]
    pub struct ResolvedField {
        pub module_idx: usize,
        pub field_idx: usize,   // 0-based index into module's field_defs
    }

    /// Per-module resolution results.
    #[derive(Debug, Clone, Default)]
    pub struct ResolvedRefs {
        pub types: HashMap<u32, ResolvedType>,    // TypeRef row index (0-based) -> resolved target
        pub methods: HashMap<u32, ResolvedMethod>, // MethodRef row index -> resolved target
        pub fields: HashMap<u32, ResolvedField>,   // FieldRef row index -> resolved target
    }

    impl ResolvedRefs {
        pub fn new() -> Self { Self::default() }
    }
    ```

    Define the Domain struct:
    ```rust
    /// A domain holds all loaded modules and provides cross-module resolution.
    pub struct Domain {
        pub modules: Vec<LoadedModule>,
    }

    impl Domain {
        pub fn new() -> Self {
            Domain { modules: Vec::new() }
        }

        /// Add a module to the domain. Returns the module's index.
        pub fn add_module(&mut self, module: writ_module::Module) -> Result<usize, RuntimeError> {
            let loaded = LoadedModule::from_module(module)?;
            let idx = self.modules.len();
            self.modules.push(loaded);
            Ok(idx)
        }

        /// Resolve all cross-module references across all loaded modules.
        pub fn resolve_refs(&mut self) -> Result<(), RuntimeError> {
            // For each module, resolve its TypeRef, MethodRef, FieldRef entries
            let module_count = self.modules.len();
            for src_idx in 0..module_count {
                let resolved = self.resolve_module_refs(src_idx)?;
                self.modules[src_idx].resolved_refs = resolved;
            }
            Ok(())
        }
    }
    ```

    ```rust
    fn find_module_by_name(&self, name: &str) -> Option<usize> {
        for (idx, m) in self.modules.iter().enumerate() {
            let mod_name = writ_module::heap::read_string(
    ```

    Helper: find TypeDef by (namespace, name) in a target module:

    ```rust
    fn find_type_def_by_name(module: &writ_module::Module, namespace: &str, name: &str) -> Option<usize> {
        for (idx, td) in module.type_defs.iter().enumerate() {
            let td_name = writ_module::heap::read_string(&module.string_heap, td.name).unwrap_or("");
            let td_ns = writ_module::heap::read_string(&module.string_heap, td.namespace).unwrap_or("");
            if td_name == name && td_ns == namespace {
                return Some(idx);
            }
        }
        None
    }
    ```

    Helper: find MethodDef by name within a type's method range:
    ```rust
    fn find_method_in_type(module: &writ_module::Module, type_idx: usize, method_name: &str) -> Option<usize> {
        let td = &module.type_defs[type_idx];
        let method_start = td.method_list.saturating_sub(1) as usize;
        let method_end = if type_idx + 1 < module.type_defs.len() {
            module.type_defs[type_idx + 1].method_list.saturating_sub(1) as usize
        } else {
            module.method_defs.len()
        };
        for idx in method_start..method_end {
            let md_name = writ_module::heap::read_string(&module.string_heap, module.method_defs[idx].name).unwrap_or("");
            if md_name == method_name {
                return Some(idx);
            }
        }
        None
    }
    ```

    **Step 4: Implement resolve_module_refs**

    ```rust
    fn resolve_module_refs(&self, src_idx: usize) -> Result<ResolvedRefs, RuntimeError> {
        let mut resolved = ResolvedRefs::new();
        let src_module = &self.modules[src_idx].module;

        // Resolve TypeRefs
        for (ref_idx, type_ref) in src_module.type_refs.iter().enumerate() {
            // Get the ModuleRef scope
            let scope_row_idx = type_ref.scope.row_index().saturating_sub(1) as usize;
            if scope_row_idx >= src_module.module_refs.len() {
                continue; // Invalid scope -- skip or error
            }
            let mod_ref = &src_module.module_refs[scope_row_idx];
            let target_mod_name = writ_module::heap::read_string(&src_module.string_heap, mod_ref.name)
                .map_err(|_| RuntimeError::ExecutionError("invalid ModuleRef name".into()))?;

            let target_mod_idx = self.find_module_by_name(target_mod_name)
                .ok_or_else(|| RuntimeError::ExecutionError(
                    format!("unresolved module reference: '{}'", target_mod_name)
                ))?;

            let ref_name = writ_module::heap::read_string(&src_module.string_heap, type_ref.name)
                .map_err(|_| RuntimeError::ExecutionError("invalid TypeRef name".into()))?;
            let ref_ns = writ_module::heap::read_string(&src_module.string_heap, type_ref.namespace)
                .map_err(|_| RuntimeError::ExecutionError("invalid TypeRef namespace".into()))?;

            let target_module = &self.modules[target_mod_idx].module;
            let typedef_idx = Self::find_type_def_by_name(target_module, ref_ns, ref_name)
                .ok_or_else(|| RuntimeError::ExecutionError(
                    format!("unresolved type reference: '{}::{}' in module '{}'", ref_ns, ref_name, target_mod_name)
                ))?;

            resolved.types.insert(ref_idx as u32, ResolvedType {
                module_idx: target_mod_idx,
                typedef_idx,
            });
        }

        // Resolve MethodRefs -- similar pattern
        // For each MethodRef: resolve parent token to a type, then find method by name in that type
        for (ref_idx, method_ref) in src_module.method_refs.iter().enumerate() {
            // method_ref.parent is a TypeDef or TypeRef token
            // If TypeRef: look up in resolved.types to get target module + typedef
            // If TypeDef: local type in same module
            // Then find method by name in the resolved type's method range
            let (target_mod_idx, type_idx) = self.resolve_parent_type(
                src_idx, method_ref.parent, &resolved
            )?;

            let method_name = writ_module::heap::read_string(&src_module.string_heap, method_ref.name)
                .map_err(|_| RuntimeError::ExecutionError("invalid MethodRef name".into()))?;

            let target_module = &self.modules[target_mod_idx].module;
            let method_idx = Self::find_method_in_type(target_module, type_idx, method_name)
                .ok_or_else(|| RuntimeError::ExecutionError(
                    format!("unresolved method reference: '{}' on type index {}", method_name, type_idx)
                ))?;

            resolved.methods.insert(ref_idx as u32, ResolvedMethod {
                module_idx: target_mod_idx,
                method_idx,
            });
        }

        // Resolve FieldRefs -- similar to MethodRefs
        for (ref_idx, field_ref) in src_module.field_refs.iter().enumerate() {
            let (target_mod_idx, type_idx) = self.resolve_parent_type(
                src_idx, field_ref.parent, &resolved
            )?;

            let field_name = writ_module::heap::read_string(&src_module.string_heap, field_ref.name)
                .map_err(|_| RuntimeError::ExecutionError("invalid FieldRef name".into()))?;

            let target_module = &self.modules[target_mod_idx].module;
            let field_idx = Self::find_field_in_type(target_module, type_idx, field_name)
                .ok_or_else(|| RuntimeError::ExecutionError(
                    format!("unresolved field reference: '{}' on type index {}", field_name, type_idx)
                ))?;

            resolved.fields.insert(ref_idx as u32, ResolvedField {
                module_idx: target_mod_idx,
                field_idx,
            });
        }

        Ok(resolved)
    }
    ```

    **Step 5: Implement resolve_parent_type helper**

    This resolves a parent MetadataToken (from MethodRef.parent or FieldRef.parent) to a (module_idx, typedef_idx) pair:
    ```rust
    fn resolve_parent_type(
        &self,
        src_idx: usize,
        parent: MetadataToken,
        resolved: &ResolvedRefs,
    ) -> Result<(usize, usize), RuntimeError> {
        let table_id = parent.table_id();
        let row_idx = parent.row_index().saturating_sub(1); // 0-based

        match table_id {
            2 => {
                // TypeDef -- local type in same module
                Ok((src_idx, row_idx as usize))
            }
            3 => {
                // TypeRef -- already resolved
                let rt = resolved.types.get(&row_idx)
                    .ok_or_else(|| RuntimeError::ExecutionError(
                        format!("parent TypeRef {} not yet resolved", row_idx)
                    ))?;
                Ok((rt.module_idx, rt.typedef_idx))
            }
            _ => Err(RuntimeError::ExecutionError(
                format!("unexpected parent token table ID: {}", table_id)
            ))
        }
    }
    ```

    ```rust
    use std::collections::HashMap;

    /// Globally unique key for dispatch table lookup.
    #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
    pub struct DispatchKey {
        pub type_key: u32,     // (module_idx << 16) | typedef_row_idx
        pub contract_key: u32, // (module_idx << 16) | contractdef_row_idx
        pub slot: u16,
    }

    /// Target of a virtual dispatch.
    #[derive(Debug, Clone, Copy)]
    pub enum DispatchTarget {
        /// IL method body in a specific module.
        Method { module_idx: usize, method_idx: usize },
        /// Runtime-provided native implementation.
        Intrinsic(IntrinsicId),
    }

    /// All intrinsic method implementations.
    #[derive(Debug, Clone, Copy, PartialEq, Eq)]
    pub enum IntrinsicId {
        // Int (13)
        IntAdd, IntSub, IntMul, IntDiv, IntMod, IntNeg, IntNot,
        IntEq, IntOrd, IntBitAnd, IntBitOr, IntIntoFloat, IntIntoString,
        // Float (10)
        FloatAdd, FloatSub, FloatMul, FloatDiv, FloatMod, FloatNeg,
        FloatEq, FloatOrd, FloatIntoInt, FloatIntoString,
        // Bool (3)
        BoolEq, BoolNot, BoolIntoString,
        // String (6)
        StringAdd, StringEq, StringOrd, StringIndexChar, StringIndexRange, StringIntoString,
        // Array (4)
        ArrayIndex, ArrayIndexSet, ArraySlice, ArrayIterable,
    }

    /// The dispatch table for O(1) contract method resolution.
    pub struct DispatchTable {
        table: HashMap<DispatchKey, DispatchTarget>,
    }

    impl DispatchTable {
        pub fn new() -> Self {
            DispatchTable { table: HashMap::new() }
        }

        pub fn insert(&mut self, key: DispatchKey, target: DispatchTarget) {
            self.table.insert(key, target);
        }

        pub fn get(&self, key: &DispatchKey) -> Option<&DispatchTarget> {
            self.table.get(key)
        }

        pub fn len(&self) -> usize {
            self.table.len()
        }
    }
    ```

    **Step 2: Add build_dispatch_table to domain.rs**

    ```rust
    use crate::dispatch::{DispatchKey, DispatchTable, DispatchTarget, IntrinsicId};

    impl Domain {
        /// Build the dispatch table from ImplDef rows across all loaded modules.
        pub fn build_dispatch_table(&self) -> DispatchTable {
            let mut table = DispatchTable::new();

            for (mod_idx, loaded) in self.modules.iter().enumerate() {
                let module = &loaded.module;

                for (impl_idx, impl_def) in module.impl_defs.iter().enumerate() {
                    // Resolve type_token to a type_key
                    let type_key = self.resolve_type_key(mod_idx, impl_def.type_token);
                    // Resolve contract token to a contract_key
                    let contract_key = self.resolve_contract_key(mod_idx, impl_def.contract);

                    // Find the method range for this ImplDef
                    let method_start = impl_def.method_list.saturating_sub(1) as usize;
                    let method_end = if impl_idx + 1 < module.impl_defs.len() {
                        module.impl_defs[impl_idx + 1].method_list.saturating_sub(1) as usize
                    } else {
                        module.method_defs.len()
                    };

                    // For each method in this impl, find the contract method slot
                    // and insert into the dispatch table
                    for method_idx in method_start..method_end {
                        let method_def = &module.method_defs[method_idx];
                        let slot = self.find_slot_for_impl_method(mod_idx, impl_def.contract, method_idx);

                        let target = if method_def.flags & 0x80 != 0 {
                            // Intrinsic method -- resolve to IntrinsicId
                            let type_name = self.get_type_name(mod_idx, impl_def.type_token);
                            let method_name = writ_module::heap::read_string(
                                &module.string_heap, method_def.name
                            ).unwrap_or("");
                            let intrinsic = resolve_intrinsic_id(&type_name, method_name);
                            DispatchTarget::Intrinsic(intrinsic)
                        } else {
                            DispatchTarget::Method { module_idx: mod_idx, method_idx }
                        };

                        table.insert(DispatchKey { type_key, contract_key, slot }, target);
                    }
                }
            }

            table
        }

        fn resolve_type_key(&self, mod_idx: usize, token: MetadataToken) -> u32 {
            // Encode as (module_idx << 16) | row_idx
            let row = token.row_index().saturating_sub(1);
            let table_id = token.table_id();
            if table_id == 2 {
                // Local TypeDef
                ((mod_idx as u32) << 16) | row
            } else if table_id == 3 {
                // TypeRef -- resolve to target module + TypeDef
                if let Some(resolved) = self.modules[mod_idx].resolved_refs.types.get(&row) {
                    ((resolved.module_idx as u32) << 16) | (resolved.typedef_idx as u32)
                } else {
                    u32::MAX // Should not happen after resolve_refs
                }
            } else {
                u32::MAX
            }
        }

        fn resolve_contract_key(&self, mod_idx: usize, token: MetadataToken) -> u32 {
            // ContractDef tokens use table ID 10
            let row = token.row_index().saturating_sub(1);
            let table_id = token.table_id();
            if table_id == 10 {
                // Local ContractDef
                ((mod_idx as u32) << 16) | row
            } else if table_id == 3 {
                // TypeRef pointing to a contract in another module
                // (contracts are referenced via TypeRef with ContractDef target)
                if let Some(resolved) = self.modules[mod_idx].resolved_refs.types.get(&row) {
                    ((resolved.module_idx as u32) << 16) | (resolved.typedef_idx as u32)
                } else {
                    u32::MAX
                }
            } else {
                u32::MAX
            }
        }
    }
    ```

    **Step 2: Implement CALL_VIRT dispatch**

    Replace the stub:
    ```rust
    Instruction::CallVirt { r_dst, r_obj, contract_idx, slot, r_base, argc } => {
        let frame = task.call_stack.last().unwrap();
        let obj_val = frame.registers[r_obj as usize];

        // Determine type_key from the object value
        let type_key = resolve_runtime_type_key(obj_val, heap, modules, current_module_idx);

        // Resolve contract_key from the contract_idx (which is a metadata token in the current module)
        let contract_key = resolve_contract_key_from_idx(contract_idx, modules, current_module_idx);

        let key = DispatchKey { type_key, contract_key, slot };

        match dispatch_table.get(&key) {
            Some(DispatchTarget::Method { module_idx, method_idx }) => {
                // Cross-module method call
                let target_module = &modules[*module_idx];
                if *method_idx >= target_module.decoded_bodies.len() {
                    return ExecutionResult::Crash(format!(
                        "CALL_VIRT: method index {} out of range in module {}",
                        method_idx, module_idx
                    ));
                }
                let reg_count = target_module.module.method_bodies[*method_idx].register_types.len();

                let caller = task.call_stack.last().unwrap();
                let mut args = Vec::with_capacity(argc as usize);
                for i in 0..argc as usize {
                    args.push(caller.registers[r_base as usize + i]);
                }

                let mut new_frame = crate::frame::CallFrame::new(*method_idx, reg_count, r_dst);
                for (i, arg) in args.into_iter().enumerate() {
                    if i < new_frame.registers.len() {
                        new_frame.registers[i] = arg;
                    }
                }

                task.call_stack.push(new_frame);
                // TODO: track current_module_idx per frame for cross-module calls
                ExecutionResult::Continue
            }
            Some(DispatchTarget::Intrinsic(id)) => {
                execute_intrinsic(*id, task, r_dst, r_obj, r_base, argc, heap)
            }
            None => {
                ExecutionResult::Crash(format!(
                    "CALL_VIRT: no implementation for type_key={}, contract_key={}, slot={}",
                    type_key, contract_key, slot
                ))
            }
        }
    }
    ```

    **Step 3: Implement resolve_runtime_type_key**

    Maps a runtime Value to a type_key:
    ```rust
    fn resolve_runtime_type_key(
        val: Value,
        heap: &dyn GcHeap,
        modules: &[LoadedModule],
        current_module_idx: usize,
    ) -> u32 {
        match val {
            Value::Int(_) => {
                // Find Int pseudo-TypeDef in virtual module (module 0)
                find_type_key_by_name(modules, 0, "Int")
            }
            Value::Float(_) => find_type_key_by_name(modules, 0, "Float"),
            Value::Bool(_) => find_type_key_by_name(modules, 0, "Bool"),
            Value::Ref(href) => {
                // For heap objects, determine type from the object
                match heap.get_object(href) {
                    Ok(HeapObject::String(_)) => find_type_key_by_name(modules, 0, "String"),
                    Ok(HeapObject::Array { .. }) => find_type_key_by_name(modules, 0, "Array"),
                    Ok(HeapObject::Struct { .. }) => {
                        // For user-defined structs, we need the type_idx stored on the struct
                        // For now, use current module context
                        // TODO: Structs need type tagging for full dispatch
                        u32::MAX
                    }
                    Ok(HeapObject::Boxed(inner)) => {
                        // Boxed value -- dispatch based on inner value type
                        resolve_runtime_type_key(*inner, heap, modules, current_module_idx)
                    }
                    _ => u32::MAX,
                }
            }
            Value::Entity(_) => find_type_key_by_name(modules, 0, "Entity"),
            Value::Void => u32::MAX,
        }
    }

    fn find_type_key_by_name(modules: &[LoadedModule], mod_idx: usize, name: &str) -> u32 {
        if mod_idx >= modules.len() { return u32::MAX; }
        let module = &modules[mod_idx].module;
        for (idx, td) in module.type_defs.iter().enumerate() {
            if let Ok(td_name) = writ_module::heap::read_string(&module.string_heap, td.name) {
                if td_name == name {
                    return ((mod_idx as u32) << 16) | (idx as u32);
                }
            }
        }
        u32::MAX
    }
    ```

    **Step 4: Implement execute_intrinsic**

    Execute the intrinsic operation and store the result in r_dst:
    ```rust
    fn execute_intrinsic(
        id: IntrinsicId,
        task: &mut Task,
        r_dst: u16,
        r_obj: u16,
        r_base: u16,
        argc: u16,
        heap: &mut dyn GcHeap,
    ) -> ExecutionResult {
        let frame = task.call_stack.last().unwrap();

        match id {
            IntrinsicId::IntAdd => {
                let a = extract_int(&frame.registers[r_obj as usize]);
                let b = if argc > 1 { extract_int(&frame.registers[r_base as usize + 1]) }
                        else { extract_int(&frame.registers[r_base as usize]) };
                let frame = task.call_stack.last_mut().unwrap();
                frame.registers[r_dst as usize] = Value::Int(a.wrapping_add(b));
                ExecutionResult::Continue
            }
            // ... similar for all other intrinsics
            // Int arithmetic: IntSub, IntMul, IntDiv (check div-by-zero), IntMod, IntNeg, IntNot
            // Int comparison: IntEq, IntOrd
            // Int bitwise: IntBitAnd, IntBitOr
            // Int conversion: IntIntoFloat (I2F), IntIntoString (I2S)
            // Float arithmetic/comparison/conversion analogous
            // Bool: BoolEq, BoolNot, BoolIntoString
            // String: StringAdd (StrConcat), StringEq (CmpEqS), StringOrd, StringIndexChar, StringIndexRange, StringIntoString
            // Array: ArrayIndex (ArrayLoad), ArrayIndexSet (ArrayStore), ArraySlice, ArrayIterable
        }
    }
    ```

    The intrinsic implementations reuse the same logic as the existing dedicated instruction handlers (e.g., IntAdd does the same thing as AddI). Extract shared helpers where possible.

    **IMPORTANT for intrinsic arg access:** In CALL_VIRT, r_obj is the self parameter and args are in r_base..r_base+argc. For binary operators (Add, Sub, etc.), self is the left operand (r_obj) and the argument is r_base+0 (or r_base+1 depending on whether self is included in r_base per spec). Per spec: "r_base..+argc are all args (r_obj should be r_base when self is the first arg)". So r_obj == r_base for methods with self, and the "other" argument is at r_base+1.

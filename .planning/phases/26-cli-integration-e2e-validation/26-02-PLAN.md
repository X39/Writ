impl LoweringError {
    pub fn to_diagnostic(&self, file_id: FileId) -> Diagnostic {
        match self {
            LoweringError::UnknownSpeaker { name, span } => {
                DiagnosticBuilder::error("L0001", format!("unknown speaker `{name}`"))
                    .file(file_id)
                    .span(*span)
                    .label(format!("speaker `{name}` is not defined"))
                    .help("speakers must be declared as [Singleton] entities with a Speaker component")
                    .build()
            }
            LoweringError::NonTerminalTransition { span } => {
                DiagnosticBuilder::error("L0002", "dialogue transition must be last statement".to_string())
                    .file(file_id)
                    .span(*span)
                    .label("this `->` transition is not the final statement in its block")
                    .build()
            }
            LoweringError::DuplicateLocKey { key, first_span, second_span } => {
                DiagnosticBuilder::error("L0003", format!("duplicate localization key `{key}`"))
                    .file(file_id)
                    .span(*second_span)
                    .label("duplicate key here")
                    .secondary(file_id, *first_span, "first defined here")
                    .build()
            }
            LoweringError::ConflictingComponentMethod { method, first_component, second_component, span } => {
                DiagnosticBuilder::error("L0004", format!("conflicting component method `{method}`"))
                    .file(file_id)
                    .span(*span)
                    .label(format!("method `{method}` exists in both `{first_component}` and `{second_component}`"))
                    .build()
            }
            LoweringError::DuplicateUseClause { component, entity, span } => {
                DiagnosticBuilder::error("L0005", format!("duplicate `use {component}` in entity `{entity}`"))
                    .file(file_id)
                    .span(*span)
                    .label(format!("`{component}` is already used"))
                    .build()
            }
            LoweringError::DuplicateProperty { property, entity, span } => {
                DiagnosticBuilder::error("L0006", format!("duplicate property `{property}` in entity `{entity}`"))
                    .file(file_id)
                    .span(*span)
                    .label(format!("`{property}` is already defined"))
                    .build()
            }
            LoweringError::UnknownLifecycleEvent { event, span } => {
                DiagnosticBuilder::error("L0007", format!("unknown lifecycle event `{event}`"))
                    .file(file_id)
                    .span(*span)
                    .label(format!("`{event}` is not a valid lifecycle event"))
                    .help("valid events: create, interact, destroy")
                    .build()
            }
            LoweringError::PropertyComponentCollision { name, entity, span } => {
                DiagnosticBuilder::error("L0008", format!("property `{name}` conflicts with component `use {name}` in entity `{entity}`"))
                    .file(file_id)
                    .span(*span)
                    .label(format!("`{name}` is both a property and a component"))
                    .build()
            }
            LoweringError::Generic { message, span } => {
                DiagnosticBuilder::error("L0099", message.clone())
                    .file(file_id)
                    .span(*span)
                    .label(message.clone())
                    .build()
            }
        }
    }
}

**Step 3: Implement cmd_compile()**

Follow the `cmd_assemble` pattern. The function chains all 5 pipeline stages with error-stop discipline (each stage checks for errors before calling the next).

```rust
fn cmd_compile(input: String, output: Option<String>) -> Result<(), String> {
    let src = std::fs::read_to_string(&input)
        .map_err(|e| format!("failed to read '{}': {}", input, e))?;

    let file_id = writ_diagnostics::FileId(0);
    let sources = [(file_id, input.as_str(), src.as_str())];

    // Stage 1: Parse
    let (cst_opt, parse_errs) = writ_parser::parse(&src);
    if !parse_errs.is_empty() {
        // Parse errors from chumsky Rich type — render them manually
        for err in &parse_errs {
            // Rich errors have .span() and .to_string()
            eprintln!("parse error: {}", err);
        }
        return Err(format!("{} parse error(s)", parse_errs.len()));
    }
    let cst = cst_opt.ok_or_else(|| "parse failed: no output".to_string())?;

    // Stage 2: Lower CST -> AST
    let (ast, lower_errs) = writ_compiler::lower(cst);
    if !lower_errs.is_empty() {
        let diags: Vec<_> = lower_errs.iter().map(|e| e.to_diagnostic(file_id)).collect();
        eprint!("{}", writ_diagnostics::render_diagnostics(&diags, &sources));
        return Err(format!("{} lowering error(s)", lower_errs.len()));
    }

    // Stage 3: Name resolution
    let (resolved, resolve_diags) = writ_compiler::resolve::resolve(
        &[(file_id, &ast)],
        &[(file_id, input.as_str())],
    );
    let has_resolve_errors = resolve_diags.iter().any(|d| d.severity == writ_diagnostics::Severity::Error);
    if !resolve_diags.is_empty() {
        eprint!("{}", writ_diagnostics::render_diagnostics(&resolve_diags, &sources));
    }
    if has_resolve_errors {
        return Err(format!("resolution failed"));
    }

    // Stage 4: Type checking
    let (typed_ast, interner, type_diags) = writ_compiler::check::typecheck(
        resolved,
        &[(file_id, &ast)],
    );
    let has_type_errors = type_diags.iter().any(|d| d.severity == writ_diagnostics::Severity::Error);
    if !type_diags.is_empty() {
        eprint!("{}", writ_diagnostics::render_diagnostics(&type_diags, &sources));
    }
    if has_type_errors {
        return Err(format!("type checking failed"));
    }

    // Stage 5: IL codegen
    let bytes = writ_compiler::emit_bodies(&typed_ast, &interner).map_err(|diags| {
        eprint!("{}", writ_diagnostics::render_diagnostics(&diags, &sources));
        format!("{} codegen error(s)", diags.len())
    })?;

    // Determine output path
    let out_path = output.unwrap_or_else(|| {
        if input.ends_with(".writ") {
            input[..input.len() - 5].to_string() + ".writil"
        } else {
            input.clone() + ".writil"
        }
    });

    // Write output
    std::fs::write(&out_path, &bytes)
        .map_err(|e| format!("failed to write '{}': {}", out_path, e))?;

    eprintln!("Compiled: {out_path}");
    Ok(())
}
```

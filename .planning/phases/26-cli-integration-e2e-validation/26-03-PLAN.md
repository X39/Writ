```rust
// Direct API approach (no subprocess):
use writ_parser;
use writ_compiler;
use writ_diagnostics::FileId;

fn compile_source(src: &str) -> Result<Vec<u8>, String> {
    let (cst, parse_errs) = writ_parser::parse(src);
    if !parse_errs.is_empty() {
        return Err(format!("{} parse errors", parse_errs.len()));
    }
    let cst = cst.unwrap();
    let (ast, lower_errs) = writ_compiler::lower(cst);
    if !lower_errs.is_empty() {
        return Err(format!("{} lower errors", lower_errs.len()));
    }
    let file_id = FileId(0);
    let (resolved, resolve_diags) = writ_compiler::resolve::resolve(
        &[(file_id, &ast)],
        &[(file_id, "test.writ")],
    );
    if resolve_diags.iter().any(|d| d.severity == writ_diagnostics::Severity::Error) {
        return Err(format!("resolution errors"));
    }
    let (typed, interner, type_diags) = writ_compiler::check::typecheck(resolved, &[(file_id, &ast)]);
    if type_diags.iter().any(|d| d.severity == writ_diagnostics::Severity::Error) {
        return Err(format!("type errors"));
    }
    writ_compiler::emit_bodies(&typed, &interner)
        .map_err(|diags| format!("{} codegen errors", diags.len()))
}
```

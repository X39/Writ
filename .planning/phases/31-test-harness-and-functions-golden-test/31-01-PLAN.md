```rust
fn compile_source(src: &str) -> Result<Vec<u8>, String> {
    let src_static: &'static str = Box::leak(src.to_string().into_boxed_str());
    let file_id = FileId(0);

    let (cst_opt, parse_errs) = writ_parser::parse(src_static);
    if !parse_errs.is_empty() {
        return Err(format!("{} parse error(s): {:?}", parse_errs.len(), parse_errs.first()));
    }
    let cst = cst_opt.ok_or_else(|| "parse failed: no CST output".to_string())?;

    let (ast, lower_errs) = writ_compiler::lower(cst);
    if !lower_errs.is_empty() {
        return Err(format!("{} lowering error(s)", lower_errs.len()));
    }

    let (resolved, resolve_diags) = writ_compiler::resolve::resolve(
        &[(file_id, &ast)],
        &[(file_id, "test.writ")],
    );
    let has_resolve_errors = resolve_diags.iter().any(|d| d.severity == Severity::Error);
    if has_resolve_errors { ... }

    let (typed_ast, interner, type_diags) = writ_compiler::check::typecheck(
        resolved,
        &[(file_id, &ast)],
    );
    let has_type_errors = type_diags.iter().any(|d| d.severity == Severity::Error);
    if has_type_errors { ... }

    writ_compiler::emit_bodies(&typed_ast, &interner, &[(file_id, &ast)])
        .map_err(|diags| { ... })
}
```

**Step 3: Create `writ-golden/tests/golden_tests.rs`.**

Implement three sections:

**Section A — `compile_and_disassemble(src: &str) -> String`**

Spawns a 16MB stack thread (same pattern as `writ-cli/src/main.rs::cmd_compile`), runs the full pipeline (parse → lower → resolve → typecheck → emit_bodies), obtains `Vec<u8>`, then calls `Module::from_bytes(&bytes).unwrap()` to deserialize into a fresh `Module`, then calls `writ_assembler::disassemble(&module)` and returns the String. Panics with a descriptive message if any pipeline stage fails.

Do NOT share the in-memory module from the compiler. The round-trip through `Module::from_bytes` is the isolation boundary required by the decisions in 31-CONTEXT.md.

**Section B — `run_golden_test(name: &str)`**

```

src_path = golden_dir.join(format!("{name}.writ"))
expected_path = golden_dir.join(format!("{name}.expected"))
src = fs::read_to_string(&src_path).expect(...)
actual = compile_and_disassemble(&src)
```

If `BLESS=1` env var is set: write `actual` to `expected_path` and print `"blessed: {expected_path}"`, then return (test passes).

Otherwise: read `expected_path` (if missing, panic with "expected file not found — run BLESS=1 cargo test -p writ-golden -- {name} to create it"). Compare `expected` and `actual` byte-by-byte. If equal, test passes. If not equal:

```rust
use similar::{ChangeTag, TextDiff};
let diff = TextDiff::from_lines(&expected, &actual);
let mut diff_text = String::new();
diff_text.push_str("--- expected\n+++ actual\n");
for change in diff.iter_all_changes() {
    let prefix = match change.tag() {
        ChangeTag::Delete => "-",
        ChangeTag::Insert => "+",
        ChangeTag::Equal  => " ",
    };
    diff_text.push_str(&format!("{prefix}{change}"));
}
panic!("golden file mismatch for '{name}':\n{diff_text}\nTo update: BLESS=1 cargo test -p writ-golden");
```
